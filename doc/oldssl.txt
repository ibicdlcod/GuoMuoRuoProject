build static qt with (1.x) openssl support

./configure -static -prefix ~/Qt_static -openssl-linked -- -D OPENSSL_ROOT_DIR=~/Qt/Tools/OpenSSL/src

ninja qtbase/all
cmake –install qtbase

ninja qttools/all
cmake –install qttools
